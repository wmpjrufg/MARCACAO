Example 1
{: .label .label-yellow }

<p align = "justify">We use the <code>LINE_CHART</code> function to plot a series of fruit prices.</p>

```python
# Data
LEGEND = ['Banana', 'Apple', 'Orange', 'Pineapple', 'Mango']
LINE_STY = ['--', '-', ':', '-.', 'None']
PRICES = [[2.5, 3.0, 2.2, 4.5, 3.5],
            [2.2, 2.8, 2.4, 4.3, 3.2],
            [2.8, 3.2, 2.5, 4.0, 3.8],
            [2.6, 2.9, 2.3, 4.2, 3.6],
            [2.3, 3.1, 2.6, 4.1, 3.4]]
YEARS = list(range(2012, 2017))
COLORS = ['#000000','olive','r','yellow','green']

# Chart setup
CHART_CONFIG = {
              'NAME': 'figure5',
              'WIDTH': 0.20, 
              'HEIGHT': 0.10,
              'MARKER': 's',
              'MARKER SIZE': 3,
              'LINE WIDTH': 4,
              'LINE STYLE': LINE_STY,
              'X AXIS LABEL': 'Year',
              'X AXIS SIZE': 10,
              'Y AXIS LABEL': 'Price ($\$$)',
              'Y AXIS SIZE': 10,
              'AXISES COLOR': '#000000',
              'LABELS SIZE': 14,
              'LABELS COLOR': '#000000',
              'CHART COLOR': COLORS,
              'ON GRID?': True,
              'LOC LEGEND': 'upper left',
              'SIZE LEGEND': 12,
              'Y LOG': False,
              'X LOG': False,
              'DPI': 600, 
              'EXTENSION': 'svg'
             }

# Data statement 
DATA = {'X': YEARS, 'Y': PRICES, 'LEGEND': LEGEND}

# Call function
LINE_CHART(DATASET = DATA, PLOT_SETUP = CHART_CONFIG)
```
